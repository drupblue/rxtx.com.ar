mysql reset root password

01 Nov 2008

/etc/init.d/mysql stop
mysqld --skip-grant-tables &
mysql -u root

UPDATE mysql.user SET Password=PASSWORD('newpassword') WHERE User='root';
FLUSH PRIVILEGES;
QUIT

/etc/init.d/mysql stop
